Mint

a refreshing way to make websites.


Mint is designed for

not production websites.


Here's how it works:

import pymint as mint

class App(mint.App):
    def __init__(self):
	super(self)
        self.a = mint.Variable(0)

    def render(self):
        return mint.Element(
            "div",
            {},
            [
                mint.Element(
                    "h1", {}, [mint.Text("Home - clicked ", self.a, " times!")]
                ),
                mint.Element(
                    "button", {"@click": self.clicked}, [mint.Text("Click me!")]
                )
            ]
        )

    def clicked(self):
        self.a += 1
